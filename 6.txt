const LanguageContext = createContext<LanguageContextType | undefined>(
  undefined
);

export const LanguageProvider: React.FC<{ children: ReactNode }> = ({
  children,
}) => {
  const [language, setLanguageState] = useState<Language>("pl");

  const setLanguage = (
    newLang: Language,
    options?: { preventNavigation?: boolean }
  ) => {
    setLanguageState(newLang);
  };

  const t = (
    key: string,
    replacements?: Record<string, string | number>
  ): string => {
    let translation = translations[language]?.[key] || key;
    if (replacements && typeof translation === "string") {
      Object.keys(replacements).forEach((placeholder) => {
        const regex = new RegExp(`{${placeholder}}`, "g");
        translation = translation.replace(
          regex,
          String(replacements[placeholder])
        );
      });
    }
    return translation || key;
  };

  return (
    <LanguageContext.Provider value={{ language, setLanguage, t }}>
      {children}
    </LanguageContext.Provider>
  );
};

export const useLanguage = (): LanguageContextType => {
  const context = useContext(LanguageContext);
  if (!context) {
    throw new Error("useLanguage must be used within a LanguageProvider");
  }
  return context;
};

function ScrollRestoration() {

    const [, hashValue] = wouterLocation.split("#");
    if (!hashValue) {
      window.scrollTo(0, 0);
    } else {

        const element = document.getElementById(hashValue);
        if (element) {
          const navbarHeight =
            document.querySelector("nav")?.offsetHeight || 70;
          const yOffset = -navbarHeight - 20;
          const y =
            element.getBoundingClientRect().top + window.pageYOffset + yOffset;
          window.scrollTo({ top: y, behavior: "smooth" });

        }
      }, 100);

  }, [wouterLocation]);

function LanguageRouterLogic() {
  const [wouterLocation, navigate] = useLocation();
  const { language, setLanguage: setGlobalLanguage } = useLanguage();

  useEffect(() => {
    const currentFullPathInBrowser =
      window.location.pathname + window.location.search + window.location.hash;
    const {
      lang: langFromUrl,
      pageKey: pageKeyFromUrl,
      hashKey,
    } = findPageKeyByLocalizedPath(currentFullPathInBrowser);

    if (langFromUrl && langFromUrl !== language) {
      setGlobalLanguage(langFromUrl, { preventNavigation: true });
    } else if (!langFromUrl && currentFullPathInBrowser !== "/") {
      const defaultLang: Language = "pl";
      let targetPageKey: PageKey = "HOME";
      let targetHashKeyResolved;

      const parsedAsDefaultLang = findPageKeyByLocalizedPath(
        currentFullPathInBrowser
      );

      if (
        parsedAsDefaultLang.lang === "pl" &&
        parsedAsDefaultLang.pageKey &&
        parsedAsDefaultLang.pageKey !== "NOT_FOUND"
      ) {
        targetPageKey = parsedAsDefaultLang.pageKey;
        targetHashKeyResolved = parsedAsDefaultLang.hashKey;
      }

      const redirectTo = getLocalizedPath(
        targetPageKey,
        defaultLang,
        targetHashKeyResolved || undefined
      );

      if (wouterLocation !== redirectTo) {
        navigate(redirectTo, { replace: true });
      }
    } else if (currentFullPathInBrowser === "/") {
      if (language !== "pl") {
        setGlobalLanguage("pl", { preventNavigation: true });
      }
    }
  }, [wouterLocation, language, setGlobalLanguage, navigate]);

  return null;
}

function LanguageRouterLogic() {
  const [wouterLocation, navigate] = useLocation();
  const { language, setLanguage: setGlobalLanguage } = useLanguage();

  useEffect(() => {
    const currentFullPathInBrowser =
      window.location.pathname + window.location.search + window.location.hash;

    const {
      lang: langFromUrlEffective,
      pageKey: pageKeyFromUrl,
      hashKey,
    } = findPageKeyByLocalizedPath(currentFullPathInBrowser);

    let targetLangToSet: Language | null = null;
    let needsRedirect = false;
    let redirectTo: string | null = null;

    if (currentFullPathInBrowser === "/") {
      targetLangToSet = "pl";
      const homePathPl = getLocalizedPath("HOME", "pl");
      if (wouterLocation !== homePathPl) {
        redirectTo = homePathPl;
        needsRedirect = true;
      }
    } else if (langFromUrlEffective) {
      targetLangToSet = langFromUrlEffective;
      if (pageKeyFromUrl && pageKeyFromUrl !== "NOT_FOUND") {
        const canonicalPathForDetectedLang = getLocalizedPath(
          pageKeyFromUrl,
          langFromUrlEffective,
          hashKey || undefined
        );
        if (currentFullPathInBrowser !== canonicalPathForDetectedLang) {
          // Potencjalne przekierowanie do kanonicznej formy URL, np. usuniÄ™cie /pl/ jeÅ›li to domyÅ›lny
          // lub dodanie /en/ jeÅ›li brakuje. Ta logika musi byÄ‡ ostroÅ¼na.
          // Na razie zakÅ‚adamy, Å¼e findPageKeyByLocalizedPath i getLocalizedPath radzÄ… sobie z tym.
        }
      } else if (pageKeyFromUrl === "NOT_FOUND" || !pageKeyFromUrl) {
        // Poprawny prefiks jÄ™zyka, ale nieznana Å›cieÅ¼ka lub nie udaÅ‚o siÄ™ zmapowaÄ‡
        // Przekieruj na stronÄ™ gÅ‚Ã³wnÄ… tego jÄ™zyka
        targetLangToSet = langFromUrlEffective; // Zachowaj jÄ™zyk z URL jeÅ›li jest poprawny
        redirectTo = getLocalizedPath("HOME", targetLangToSet);
        needsRedirect = true;
      }
    } else {
      // Brak prefiksu, a nie jest to root - sprÃ³buj zinterpretowaÄ‡ jako polski lub przekieruj na polskÄ… stronÄ™ gÅ‚Ã³wnÄ…
      targetLangToSet = "pl";
      const parsedAsPl = findPageKeyByLocalizedPath(currentFullPathInBrowser);
      if (parsedAsPl.pageKey && parsedAsPl.pageKey !== "NOT_FOUND") {
        // To jest poprawna polska Å›cieÅ¼ka bez prefiksu
        targetLangToSet = "pl"; // Upewnij siÄ™
        // Nie ma potrzeby przekierowania URL, ale upewnij siÄ™, Å¼e kontekst jest PL
      } else {
        // Nieznana Å›cieÅ¼ka bez prefiksu
        redirectTo = getLocalizedPath("HOME", targetLangToSet);
        needsRedirect = true;
      }
    }

    if (targetLangToSet && targetLangToSet !== language) {
      setGlobalLanguage(targetLangToSet, { preventNavigation: true });
    }

    if (needsRedirect && redirectTo && wouterLocation !== redirectTo) {
      navigate(redirectTo, { replace: true });
    }
  }, [wouterLocation, language, setGlobalLanguage, navigate]);

  return null;
}
