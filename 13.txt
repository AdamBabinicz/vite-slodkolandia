export const getLocalizedSlug = (pageKey: PageKey, lang: Language): string => {
  const slug = slugsConfig[lang]?.[pageKey];
  if (typeof slug === "string") return slug;
  const fallbackSlug = slugsConfig[defaultLang]?.[pageKey];
  if (typeof fallbackSlug === "string") return fallbackSlug;
  return pageKey.toLowerCase().replace(/_/g, "-");
};

// Zwraca ścieżkę BEZ prefiksu językowego, np. /oferta, /galeria
// Używane przez <Route path={...}> gdy <WouterRouter base="/:lang?">
export function getInternalRoutePath(pageKey: PageKey, lang: Language): string {
  if (pageKey === PAGE_KEYS.HOME) return "/";
  return `/${getLocalizedSlug(pageKey, lang)}`;
}

// Generuje pełną ścieżkę URL widoczną w przeglądarce (z prefiksem językowym, jeśli nie defaultLang)
export function getLocalizedPath(
  pageKey: PageKey,
  lang: Language,
  hashPageKey?: PageKey // PageKey, którego slug będzie hashem
): string {
  const langPrefix = lang === defaultLang ? "" : `/${lang}`;
  let basePathToUse: string;

  if (pageKey === PAGE_KEYS.HOME) {
    basePathToUse = langPrefix || "/"; // Dla PL: "/", dla EN: "/en"
  } else {
    const slug = getLocalizedSlug(pageKey, lang);
    basePathToUse = `${langPrefix}/${slug}`; // Dla PL: "/oferta", dla EN: "/en/offer"
  }

  let fullPath = basePathToUse;

  if (hashPageKey) {
    const hashSlugValue = getLocalizedSlug(hashPageKey, lang);
    if (hashSlugValue) fullPath += `#${hashSlugValue}`;
  }
  return fullPath;
}

// Wersja FPLP v12.2 (FindPageKeyByLocalizedPath) - drobne usprawnienia logowania
export function findPageKeyByLocalizedPath(currentFullPath: string): {
  pageKey: PageKey | null;
  lang: Language | null;
  hashKey: PageKey | null;
  hashSlug: string | null;
  basePathWithoutLang: string;
} {
  console.log(
    `%c[FPLP v12.2] Input currentFullPath: "${currentFullPath}"`,
    "color: blue;"
  );
  const [pathAndQuery, hashValueFromUrl] = currentFullPath.split("#");
  const pathOnly = pathAndQuery.split("?")[0];

  let detectedLang: Language = defaultLang;
  let slugToAnalyze: string;
  let basePathWithoutLang: string;

  const pathSegments = pathOnly.split("/").filter(Boolean);

  if (
    pathSegments.length > 0 &&
    supportedLngs.includes(pathSegments[0] as Language)
  ) {
    detectedLang = pathSegments[0] as Language;
    slugToAnalyze = pathSegments.slice(1).join("/") || "";
    basePathWithoutLang = `/${slugToAnalyze}`.replace(/\/+$/, "") || "/";
  } else {
    detectedLang = defaultLang;
    slugToAnalyze = pathSegments.join("/") || "";
    basePathWithoutLang = `/${slugToAnalyze}`.replace(/\/+$/, "") || "/";
  }

  console.log(
    `  [FPLP v12.2] For detectedLang "${detectedLang}": slugToAnalyze="${slugToAnalyze}", basePathWithoutLang="${basePathWithoutLang}", hashValue="${
      hashValueFromUrl || "null"
    }"`
  );

  let foundPageKey: PageKey | null = null;
  for (const keyCandidate in slugsConfig[detectedLang]) {
    const key = keyCandidate as PageKey;
    if (
      Object.prototype.hasOwnProperty.call(slugsConfig[detectedLang], key) &&
      Object.values(PAGE_KEYS).includes(key) &&
      !key.startsWith("OFFER_") &&
      !key.startsWith("GALLERY_PHOTO_")
    ) {
      if (slugsConfig[detectedLang][key] === slugToAnalyze) {
        foundPageKey = key;
        break;
      }
    }
  }
  if (
    !foundPageKey &&
    slugToAnalyze === "" &&
    slugsConfig[detectedLang][PAGE_KEYS.HOME] === ""
  ) {
    foundPageKey = PAGE_KEYS.HOME;
  }

  if (!foundPageKey) {
    foundPageKey = PAGE_KEYS.NOT_FOUND;
  }

  let foundHashKey: PageKey | null = null;
  if (hashValueFromUrl) {
    for (const keyCandidate in slugsConfig[detectedLang]) {
      const key = keyCandidate as PageKey;
      if (
        Object.prototype.hasOwnProperty.call(slugsConfig[detectedLang], key) &&
        Object.values(PAGE_KEYS).includes(key)
      ) {
        if (slugsConfig[detectedLang][key] === hashValueFromUrl) {
          foundHashKey = key;
          break;
        }
      }
    }
    if (!foundHashKey) {
      console.warn(
        `  [FPLP v12.2] No PageKey found whose slug matches the hashSlug "${hashValueFromUrl}" for lang "${detectedLang}".`
      );
    }
  }

  const langToReturn =
    foundPageKey === PAGE_KEYS.NOT_FOUND ? null : detectedLang;

  console.log(
    `%c  [FPLP v12.2] Returning: pageKey="${foundPageKey}", lang="${langToReturn}", hashKey="${
      foundHashKey || "null"
    }", hashSlug="${
      hashValueFromUrl || null
    }", basePathWithoutLang="${basePathWithoutLang}"`,
    "color: green;"
  );
  return {
    pageKey: foundPageKey,
    lang: langToReturn,
    hashKey: foundHashKey,
    hashSlug: hashValueFromUrl || null,
    basePathWithoutLang: basePathWithoutLang,
  };
}

export const getLocalizedSlug = (pageKey: PageKey, lang: Language): string => {
  const slug = slugsConfig[lang]?.[pageKey];
  if (typeof slug === "string") return slug;
  const fallbackSlug = slugsConfig[defaultLang]?.[pageKey];
  if (typeof fallbackSlug === "string") return fallbackSlug;
  return pageKey.toLowerCase().replace(/_/g, "-");
};

export function getInternalRoutePath(pageKey: PageKey, lang: Language): string {
  if (pageKey === PAGE_KEYS.HOME) return "/";
  return `/${getLocalizedSlug(pageKey, lang)}`;
}

export function getLocalizedPath(
  pageKey: PageKey,
  lang: Language,
  hashPageKey?: PageKey
): string {
  const langPrefix = lang === defaultLang ? "" : `/${lang}`;
  let basePathKey: PageKey = pageKey;
  let actualHashKeyForSlug: PageKey | undefined = hashPageKey;

  if (pageKey.startsWith("OFFER_")) {
    basePathKey = PAGE_KEYS.OFFER;
    actualHashKeyForSlug = pageKey;
  } else if (pageKey.startsWith("GALLERY_PHOTO_")) {
    basePathKey = PAGE_KEYS.GALLERY;
    actualHashKeyForSlug = pageKey;
  }

  let baseSlugPart;
  if (basePathKey === PAGE_KEYS.HOME) {
    baseSlugPart = "";
  } else {
    baseSlugPart = `/${getLocalizedSlug(basePathKey, lang)}`;
  }

  let fullPath = `${langPrefix}${baseSlugPart}`;

  if (basePathKey === PAGE_KEYS.HOME && lang === defaultLang) fullPath = "/";
  if (
    basePathKey === PAGE_KEYS.HOME &&
    lang !== defaultLang &&
    fullPath === `/${lang}/`
  ) {
    fullPath = `/${lang}`;
  }
  if (basePathKey !== PAGE_KEYS.HOME && fullPath === "" && langPrefix) {
    fullPath = langPrefix; // np. /en jeśli slug był pusty
  } else if (basePathKey !== PAGE_KEYS.HOME && fullPath === "") {
    fullPath = "/"; // Fallback dla defaultLang, jeśli slug był pusty
  }

  if (actualHashKeyForSlug) {
    const hashSlugValue = getLocalizedSlug(actualHashKeyForSlug, lang);
    if (hashSlugValue) fullPath += `#${hashSlugValue}`;
  }
  return fullPath;
}

export function findPageKeyByLocalizedPath(currentFullPath: string): {
  pageKey: PageKey | null;
  lang: Language; // Zawsze zwraca wykryty język
  hashKey: PageKey | null;
  hashSlug: string | null;
  basePathWithoutLang: string;
} {
  console.log(
    `%c[FPLP v12.4] Input currentFullPath: "${currentFullPath}"`,
    "color: blue;"
  );
  const [pathAndQuery, hashValueFromUrl] = currentFullPath.split("#");
  const pathOnly = pathAndQuery.split("?")[0];

  let detectedLang: Language = defaultLang;
  let slugSegmentsForPathMatching: string[];

  const pathSegments = pathOnly.split("/").filter(Boolean);

  if (
    pathSegments.length > 0 &&
    supportedLngs.includes(pathSegments[0] as Language)
  ) {
    detectedLang = pathSegments[0] as Language;
    slugSegmentsForPathMatching = pathSegments.slice(1);
    console.log(`  [FPLP v12.4] Detected lang from prefix: "${detectedLang}"`);
  } else {
    detectedLang = defaultLang;
    slugSegmentsForPathMatching = pathSegments;
    console.log(
      `  [FPLP v12.4] No lang prefix. Assumed lang: "${detectedLang}"`
    );
  }

  const slugToAnalyze = slugSegmentsForPathMatching.join("/") || ""; // To jest slug BEZ języka
  const basePathWithoutLang = `/${slugToAnalyze}`.replace(/\/+$/, "") || "/";

  console.log(
    `  [FPLP v12.4] For detectedLang "${detectedLang}": slugToAnalyze="${slugToAnalyze}", basePathWithoutLang="${basePathWithoutLang}", hashValue="${
      hashValueFromUrl || "null"
    }"`
  );

  let foundPageKey: PageKey | null = null;
  for (const keyCandidate in slugsConfig[detectedLang]) {
    const key = keyCandidate as PageKey;
    if (
      Object.prototype.hasOwnProperty.call(slugsConfig[detectedLang], key) &&
      Object.values(PAGE_KEYS).includes(key) &&
      !key.startsWith("OFFER_") &&
      !key.startsWith("GALLERY_PHOTO_")
    ) {
      if (slugsConfig[detectedLang][key] === slugToAnalyze) {
        foundPageKey = key;
        break;
      }
    }
  }
  if (
    !foundPageKey &&
    slugToAnalyze === "" &&
    slugsConfig[detectedLang][PAGE_KEYS.HOME] === ""
  ) {
    foundPageKey = PAGE_KEYS.HOME;
  }

  if (!foundPageKey) {
    console.warn(
      `  [FPLP v12.4] No pageKey found for slug "${slugToAnalyze}" and lang "${detectedLang}". Setting to NOT_FOUND.`
    );
    foundPageKey = PAGE_KEYS.NOT_FOUND;
  }

  let foundHashKey: PageKey | null = null;
  if (hashValueFromUrl) {
    for (const keyCandidate in slugsConfig[detectedLang]) {
      const key = keyCandidate as PageKey;
      if (
        Object.prototype.hasOwnProperty.call(slugsConfig[detectedLang], key) &&
        Object.values(PAGE_KEYS).includes(key)
      ) {
        if (slugsConfig[detectedLang][key] === hashValueFromUrl) {
          foundHashKey = key;
          break;
        }
      }
    }
    if (!foundHashKey && foundPageKey !== PAGE_KEYS.NOT_FOUND) {
      // console.warn(`  [FPLP v12.4] No PageKey found for hashSlug "${hashValueFromUrl}" (lang "${detectedLang}").`);
    }
  }

  // Zawsze zwracaj wykryty język, nawet jeśli strona to 404.
  // LanguageRouterLogic potrzebuje tego, aby wiedzieć, dla jakiego języka wyświetlić np. stronę główną.
  const langToReturn = detectedLang;

  console.log(
    `%c  [FPLP v12.4] Returning: pageKey="${foundPageKey}", lang="${langToReturn}", hashKey="${
      foundHashKey || "null"
    }", hashSlug="${
      hashValueFromUrl || null
    }", basePathWithoutLang="${basePathWithoutLang}"`,
    "color: green;"
  );
  return {
    pageKey: foundPageKey,
    lang: langToReturn, // ZAWSZE zwraca wykryty język
    hashKey: foundHashKey,
    hashSlug: hashValueFromUrl || null,
    basePathWithoutLang: basePathWithoutLang,
  };
}

function ScrollRestoration() {
  const [wouterLocation] = useLocation();

  useEffect(() => {
    const hashValue = window.location.hash.replace("#", "");

    if (!hashValue) {
      // Proste przewinięcie do góry, jeśli nie ma hasha.
      // Można by to ulepszyć, aby scrollować tylko przy zmianie ścieżki bazowej.
      window.scrollTo(0, 0);
    } else {
      const scrollTimeoutId = setTimeout(() => {
        const element = document.getElementById(hashValue);
        if (element) {
          const navbarElement = document.querySelector("nav");
          const navbarHeight = navbarElement ? navbarElement.offsetHeight : 70; // Domyślna wysokość
          const yOffset = -(navbarHeight + 20); // Dodatkowy margines
          const y =
            element.getBoundingClientRect().top + window.pageYOffset + yOffset;
          window.scrollTo({ top: y, behavior: "smooth" });
        } else {
          console.warn(
            `[ScrollRestoration] Element with id "${hashValue}" not found for scrolling.`
          );
        }
      }, 100); // Opóźnienie, aby DOM był gotowy
      return () => clearTimeout(scrollTimeoutId);
    }
  }, [wouterLocation]);
  return null;
}

function LanguageRouterLogic() {
  const [wouterLocation, navigate] = useLocation();
  const {
    language,
    setLanguage: setGlobalLanguage,
    isLanguageInitialized,
  } = useLanguage();

  const effectUrlSyncRunCountRef = useRef(0);
  const effectLangChangeNavRunCountRef = useRef(0);

  const normalizePathForComparison = (p: string) => {
    const [pathPart, hashPart] = p.split("#");
    // Usuń końcowe slashe, obsłuż "" (np. dla root path) jako "/"
    const normalizedPath = pathPart.replace(/\/+$/, "") || "/";
    return hashPart ? `${normalizedPath}#${hashPart}` : normalizedPath;
  };

  useEffect(() => {
    if (!isLanguageInitialized) {
      console.log("[LRL_v12.5_URL_SYNC] Waiting for language init...");
      return;
    }

    effectUrlSyncRunCountRef.current += 1;
    const runId = effectUrlSyncRunCountRef.current;
    const currentBrowserFullPath =
      window.location.pathname + window.location.search + window.location.hash;

    if (runId > 10) {
      console.error(
        `[LRL_v12.5_URL_SYNC #${runId}] FATAL LOOP. Path: "${currentBrowserFullPath}", Lang: "${language}". Halting.`
      );
      effectUrlSyncRunCountRef.current = 0;
      return;
    }
    console.log(
      `%c[LRL_v12.5_URL_SYNC #${runId}] START. BrowserPath: "${currentBrowserFullPath}", WouterLoc: "${wouterLocation}", GlobalLang: "${language}"`,
      "color: purple; font-weight: bold;"
    );

    const {
      pageKey: parsedPageKey,
      lang: langDetectedInUrl,
      hashKey: parsedHashKey,
    } = findPageKeyByLocalizedPath(currentBrowserFullPath);

    let langToUseForCanonicalPath = language;
    let languageContextShouldChange = false;

    if (langDetectedInUrl) {
      if (langDetectedInUrl !== language) {
        console.warn(
          `  [LRL_v12.5_URL_SYNC #${runId}] ACTION: Context lang ("${language}") differs from lang detected in URL ("${langDetectedInUrl}"). Updating context lang.`
        );
        setGlobalLanguage(langDetectedInUrl, { preventNavigation: true });
        languageContextShouldChange = true;
        // langToUseForCanonicalPath zostanie ustawione na langDetectedInUrl w następnym przebiegu tego efektu
        // lub obsłużone przez efekt LANG_NAV
      } else {
        langToUseForCanonicalPath = langDetectedInUrl;
      }
    } else {
      // Brak języka w URL, FPLP zwróciło defaultLang jako detectedLang
      if (language !== defaultLang) {
        console.warn(
          `  [LRL_v12.5_URL_SYNC #${runId}] ACTION: No lang in URL for path. Global lang ("${language}") is not default. Updating context to default ("${defaultLang}").`
        );
        setGlobalLanguage(defaultLang, { preventNavigation: true });
        languageContextShouldChange = true;
      }
      langToUseForCanonicalPath = defaultLang;
    }

    if (languageContextShouldChange) {
      console.log(
        `  [LRL_v12.5_URL_SYNC #${runId}] Language context was changed. LANG_NAV effect will handle potential navigation. ENDING URL_SYNC run now.`
      );
      // Nie resetuj licznika pętli tutaj, pozwól drugiemu efektowi dokończyć.
      // Zakończenie tego efektu teraz jest kluczowe.
      return;
    }

    // Jeśli doszliśmy tutaj, język kontekstu jest już zsynchronizowany z tym, co powinno być na podstawie URL
    // lub URL był dla defaultLang, a kontekst też był defaultLang.
    // Używamy `language` (z kontekstu) do generowania kanonicznej ścieżki.
    console.log(
      `  [LRL_v12.5_URL_SYNC #${runId}] Lang for canonical path check: "${language}" (Context and URL lang are in sync or URL implies default)`
    );

    if (parsedPageKey && parsedPageKey !== PAGE_KEYS.NOT_FOUND) {
      const canonicalPath = getLocalizedPath(
        parsedPageKey,
        language, // Użyj zsynchronizowanego języka kontekstu
        parsedHashKey || undefined
      );

      const normalizedCurrent = normalizePathForComparison(
        currentBrowserFullPath
      );
      const normalizedCanonical = normalizePathForComparison(canonicalPath);

      console.log(
        `  [LRL_v12.5_URL_SYNC Effect #${runId}] For PageKey: '${parsedPageKey}', Lang: '${language}', HashKey: '${parsedHashKey}'`
      );
      console.log(
        `    Current Path (Browser): "${currentBrowserFullPath}" (Normalized: "${normalizedCurrent}")`
      );
      console.log(
        `    Canonical Path (Generated): "${canonicalPath}" (Normalized: "${normalizedCanonical}")`
      );

      if (normalizedCurrent !== normalizedCanonical) {
        console.warn(
          `  [LRL_v12.5_URL_SYNC Effect #${runId}] ACTION: Path MISMATCH. NAVIGATING to "${canonicalPath}".`
        );
        navigate(canonicalPath, { replace: true });
        return; // Nawigacja wywoła ten efekt ponownie, nie resetuj licznika tutaj.
      } else {
        console.log(
          `  [LRL_v12.5_URL_SYNC Effect #${runId}] Path is canonical. Resetting counter.`
        );
        effectUrlSyncRunCountRef.current = 0;
      }
    } else {
      const homePath = getLocalizedPath(PAGE_KEYS.HOME, language); // Użyj zsynchronizowanego języka kontekstu
      const normalizedCurrentHomeCheck = normalizePathForComparison(
        currentBrowserFullPath
      );
      const normalizedHomePath = normalizePathForComparison(homePath);

      if (normalizedCurrentHomeCheck !== normalizedHomePath) {
        console.warn(
          `  [LRL_v12.5_URL_SYNC Effect #${runId}] Page NOT FOUND or invalid. Redirecting to home for lang "${language}": "${homePath}"`
        );
        navigate(homePath, { replace: true });
        return;
      } else {
        console.log(
          `  [LRL_v12.5_URL_SYNC Effect #${runId}] Page NOT_FOUND, but already at determined home path. Resetting counter.`
        );
        effectUrlSyncRunCountRef.current = 0;
      }
    }

    if (effectUrlSyncRunCountRef.current !== 0) {
      console.log(
        `  [LRL_v12.5_URL_SYNC Effect #${runId}] No navigation taken by URL_SYNC, counter: ${effectUrlSyncRunCountRef.current}.`
      );
    }
    console.log(
      `%c[LRL_v12.5_URL_SYNC Effect #${runId} END] Final counter: ${effectUrlSyncRunCountRef.current}`,
      "color: teal; font-weight: bold;"
    );
  }, [
    wouterLocation,
    language,
    isLanguageInitialized,
    setGlobalLanguage,
    navigate,
  ]);

  useEffect(() => {
    if (!isLanguageInitialized) {
      console.log("[LRL_v12.5_LANG_NAV] Waiting for language init...");
      return;
    }

    effectLangChangeNavRunCountRef.current += 1;
    const runId = effectLangChangeNavRunCountRef.current;

    if (runId > 5) {
      console.error(`[LRL_v12.5_LANG_NAV #${runId}] FATAL LOOP. Halting.`);
      effectLangChangeNavRunCountRef.current = 0;
      return;
    }
    console.log(
      `%c[LRL_v12.5_LANG_NAV #${runId}] START. Global lang (from context) is now: "${language}"`,
      "color: orange; font-weight: bold;"
    );

    const currentBrowserFullPath =
      window.location.pathname + window.location.search + window.location.hash;
    const {
      pageKey: currentPageKeyFromUrl,
      lang: langCurrentlyInUrl,
      hashKey: currentHashKeyFromUrl,
    } = findPageKeyByLocalizedPath(currentBrowserFullPath);

    // Jeśli język w URL *już* zgadza się z nowym językiem kontekstu,
    // A EFEKT URL_SYNC JUŻ SIĘ WYKONAŁ I NIE ZMIENIŁ JĘZYKA KONTEKSTU (licznik wyzerowany),
    // to prawdopodobnie nie ma potrzeby nawigować z tego efektu.
    if (
      langCurrentlyInUrl === language &&
      effectUrlSyncRunCountRef.current === 0
    ) {
      console.log(
        `  [LRL_v12.5_LANG_NAV #${runId}] Lang in URL ("${langCurrentlyInUrl}") matches context lang ("${language}") and URL_SYNC is stable. No navigation by LANG_NAV. Resetting counter.`
      );
      effectLangChangeNavRunCountRef.current = 0;
      return;
    }

    const targetPageKeyForNav =
      currentPageKeyFromUrl && currentPageKeyFromUrl !== PAGE_KEYS.NOT_FOUND
        ? currentPageKeyFromUrl
        : PAGE_KEYS.HOME;

    const newCanonicalPathForNewLang = getLocalizedPath(
      targetPageKeyForNav,
      language, // NOWY język z kontekstu
      currentHashKeyFromUrl || undefined
    );

    const normalizedCurrent = normalizePathForComparison(
      currentBrowserFullPath
    );
    const normalizedNewCanonical = normalizePathForComparison(
      newCanonicalPathForNewLang
    );

    console.log(
      `  [LRL_v12.5_LANG_NAV #${runId}] For new context lang "${language}", targetPageKey "${targetPageKeyForNav}", currentHashKey "${currentHashKeyFromUrl}"`
    );
    console.log(
      `    Current Path (Browser): "${currentBrowserFullPath}" (Normalized: "${normalizedCurrent}")`
    );
    console.log(
      `    New Canonical Path for new context lang: "${newCanonicalPathForNewLang}" (Normalized: "${normalizedNewCanonical}")`
    );

    if (normalizedCurrent !== normalizedNewCanonical) {
      console.warn(
        `  [LRL_v12.5_LANG_NAV #${runId}] ACTION: Lang context changed. NAVIGATING to "${newCanonicalPathForNewLang}".`
      );
      navigate(newCanonicalPathForNewLang, { replace: true });
      // Nie resetuj licznika tutaj; nawigacja wywoła efekt URL_SYNC, który powinien ustabilizować i zresetować swój licznik.
      return;
    } else {
      console.log(
        `  [LRL_v12.5_LANG_NAV #${runId}] Current path is already canonical for new lang. No navigation by LANG_NAV. Resetting counter.`
      );
      effectLangChangeNavRunCountRef.current = 0;
    }
    console.log(
      `%c[LRL_v12.5_LANG_NAV Effect #${runId} END] Final counter: ${effectLangChangeNavRunCountRef.current}`,
      "color: darkorange; font-weight: bold;"
    );
  }, [language, isLanguageInitialized, navigate]);

  return null;
}

function LanguageAwareRouter() {
  const { language, isLanguageInitialized } = useLanguage();

  if (!isLanguageInitialized) {

  const routerKey = `main-router-${base}`;

  return (
    <WouterRouter key={routerKey} base={base}>

      <AppMainContent />
    </WouterRouter>
  );

// Ustaw na true, aby włączyć szczegółowe logi debugujące z LRL
const LRL_DEBUG_MODE = false;

function ScrollRestoration() {
  const [wouterLocation] = useLocation();

  useEffect(() => {
    const hashValue = window.location.hash.replace("#", "");
    if (!hashValue) {
      window.scrollTo(0, 0);
    } else {
      const scrollTimeoutId = setTimeout(() => {
        const element = document.getElementById(hashValue);
        if (element) {
          const navbarElement = document.querySelector("nav");
          const navbarHeight = navbarElement ? navbarElement.offsetHeight : 70;
          const yOffset = -(navbarHeight + 20);
          const y =
            element.getBoundingClientRect().top + window.pageYOffset + yOffset;
          window.scrollTo({ top: y, behavior: "smooth" });
        } else {
          if (LRL_DEBUG_MODE)
            console.warn(
              `[ScrollRestoration] Element with id "${hashValue}" not found.`
            );
        }
      }, 100);
      return () => clearTimeout(scrollTimeoutId);
    }
  }, [wouterLocation]);
  return null;
}

// Wersja LRL v13.5 - Cicha
function LanguageRouterLogic() {
  const [wouterLocation, navigate] = useLocation();
  const {
    language: contextLanguage,
    setLanguage: setGlobalLanguage,
    isLanguageInitialized,
  } = useLanguage();

  const logicRunCounter = useRef(0);
  const userInitiatedLangChangeRef = useRef<Language | null>(null);

  const normalizePathForComparison = (p: string) => {
    const [pathPart, hashPart] = p.split("#");
    const normalizedPath = pathPart.replace(/\/+$/, "") || "/";
    return hashPart ? `${normalizedPath}#${hashPart}` : normalizedPath;
  };

  useEffect(() => {
    // USER_LANG_CHANGE_EFFECT
    if (!isLanguageInitialized) {
      // if (LRL_DEBUG_MODE) console.log(`%c[LRL_v13.5 USER_LANG_CHANGE_EFFECT] Waiting for init...`, "color: brown;");
      return;
    }

    if (userInitiatedLangChangeRef.current === contextLanguage) {
      return;
    }

    // if (LRL_DEBUG_MODE) console.log(`%c[LRL_v13.5 USER_LANG_CHANGE_EFFECT] Context language is now: "${contextLanguage}". Previous intent: "${userInitiatedLangChangeRef.current}"`, "color: orange; font-weight: bold;");
    userInitiatedLangChangeRef.current = contextLanguage;
    logicRunCounter.current = 0;

    const currentBrowserFullPath =
      window.location.pathname + window.location.search + window.location.hash;
    const {
      pageKey: currentPageKeyFromUrl,
      lang: langCurrentlyInUrl,
      hashKey: currentHashKeyFromUrl,
    } = findPageKeyByLocalizedPath(currentBrowserFullPath);

    // if (LRL_DEBUG_MODE) console.log(`  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] Current URL: "${currentBrowserFullPath}", langInUrl: "${langCurrentlyInUrl}", pageKeyFromUrl: "${currentPageKeyFromUrl}"`);

    if (langCurrentlyInUrl !== contextLanguage) {
      const targetPageKeyForNav =
        currentPageKeyFromUrl && currentPageKeyFromUrl !== PAGE_KEYS.NOT_FOUND
          ? currentPageKeyFromUrl
          : PAGE_KEYS.HOME;

      const newAbsolutePathTarget = getLocalizedPath(
        targetPageKeyForNav,
        contextLanguage,
        currentHashKeyFromUrl || undefined
      );

      let relativePathForNavigate = getInternalRoutePath(
        targetPageKeyForNav,
        contextLanguage
      );
      if (currentHashKeyFromUrl) {
        const hashSlug = getLocalizedSlug(
          currentHashKeyFromUrl,
          contextLanguage
        );
        if (hashSlug) relativePathForNavigate += `#${hashSlug}`;
      }

      const normalizedCurrent = normalizePathForComparison(
        currentBrowserFullPath
      );
      const normalizedNewTargetAbsolute = normalizePathForComparison(
        newAbsolutePathTarget
      );

      // if (LRL_DEBUG_MODE) console.log(`  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] New absolute target: "${newAbsolutePathTarget}", Relative for navigate: "${relativePathForNavigate}"`);

      if (normalizedCurrent !== normalizedNewTargetAbsolute) {
        console.warn(
          `  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] ACTION: Navigating to reflect new context language. Target: "${newAbsolutePathTarget}" (using relative: "${relativePathForNavigate}")`
        );
        navigate(relativePathForNavigate, { replace: true });
      } else {
        // if (LRL_DEBUG_MODE) console.log(`  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] URL already matches new context language. Clearing intent.`);
        userInitiatedLangChangeRef.current = null;
      }
    } else {
      // if (LRL_DEBUG_MODE) console.log(`  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] URL lang ("${langCurrentlyInUrl}") already matches new context language ("${contextLanguage}"). Clearing intent.`);
      userInitiatedLangChangeRef.current = null;
    }
  }, [contextLanguage, isLanguageInitialized, navigate]);

  useEffect(() => {
    // URL_SYNC_EFFECT
    if (!isLanguageInitialized) {
      return;
    }

    const currentBrowserFullPathForCheck =
      window.location.pathname + window.location.search + window.location.hash;
    const { lang: langInCurrentUrlCheck } = findPageKeyByLocalizedPath(
      currentBrowserFullPathForCheck
    );

    if (
      userInitiatedLangChangeRef.current === contextLanguage &&
      langInCurrentUrlCheck !== contextLanguage
    ) {
      // if (LRL_DEBUG_MODE) console.log(`%c[LRL_v13.5 URL_SYNC_EFFECT] User intent for lang "${contextLanguage}" is active, but URL lang is "${langInCurrentUrlCheck}". URL_SYNC waits.`, "color: gray;");
      return;
    }

    logicRunCounter.current += 1;
    const runId = logicRunCounter.current;
    const currentBrowserFullPath =
      window.location.pathname + window.location.search + window.location.hash;

    if (runId > 6) {
      console.error(
        `[LRL_v13.5 URL_SYNC_EFFECT #${runId}] FATAL LOOP. Path: "${currentBrowserFullPath}", ContextLang: "${contextLanguage}". Halting.`
      );
      logicRunCounter.current = 0;
      userInitiatedLangChangeRef.current = null;
      return;
    }

    const {
      pageKey: parsedPageKeyFromUrl,
      lang: langDetectedInUrl,
      hashKey: parsedHashKeyFromUrl,
    } = findPageKeyByLocalizedPath(currentBrowserFullPath);

    if (langDetectedInUrl !== contextLanguage) {
      console.warn(
        `  [LRL_v13.5 URL_SYNC_EFFECT #${runId}] ACTION (URL priority): Context lang ("${contextLanguage}") differs from URL lang ("${langDetectedInUrl}"). Updating context.`
      );
      setGlobalLanguage(langDetectedInUrl, { preventNavigation: true });
      return;
    }

    const pageKeyToUse =
      parsedPageKeyFromUrl && parsedPageKeyFromUrl !== PAGE_KEYS.NOT_FOUND
        ? parsedPageKeyFromUrl
        : PAGE_KEYS.HOME;
    const canonicalAbsolutePath = getLocalizedPath(
      pageKeyToUse,
      contextLanguage,
      parsedHashKeyFromUrl || undefined
    );
    const normalizedCurrent = normalizePathForComparison(
      currentBrowserFullPath
    );
    const normalizedCanonical = normalizePathForComparison(
      canonicalAbsolutePath
    );

    if (normalizedCurrent !== normalizedCanonical) {
      let reason = "Path MISMATCH";
      if (
        parsedPageKeyFromUrl === PAGE_KEYS.NOT_FOUND &&
        pageKeyToUse === PAGE_KEYS.HOME
      ) {
        reason = "Page NOT_FOUND, redirecting to HOME";
      }
      console.warn(
        `  [LRL_v13.5 URL_SYNC_EFFECT #${runId}] ACTION: ${reason}. Current: "${currentBrowserFullPath}", Canonical: "${canonicalAbsolutePath}". NAVIGATING.`
      );
      let relativePathForNavigate = getInternalRoutePath(
        pageKeyToUse,
        contextLanguage
      );
      if (parsedHashKeyFromUrl) {
        const hashSlug = getLocalizedSlug(
          parsedHashKeyFromUrl,
          contextLanguage
        );
        if (hashSlug) relativePathForNavigate += `#${hashSlug}`;
      }
      navigate(relativePathForNavigate, { replace: true });
      return;
    } else {
      logicRunCounter.current = 0;
      if (userInitiatedLangChangeRef.current === contextLanguage) {
        // if (LRL_DEBUG_MODE) console.log(`  [LRL_v13.5 URL_SYNC_EFFECT #${runId}] Path is canonical and lang intent fulfilled. Clearing userInitiatedLangChangeRef.`);
        userInitiatedLangChangeRef.current = null;
      }
    }
  }, [
    wouterLocation,
    contextLanguage,
    isLanguageInitialized,
    setGlobalLanguage,
    navigate,
  ]);

  return null;
}

function AppRoutes() {
  const { language } = useLanguage();
  return (
    <Switch>
      <Route
        path={getInternalRoutePath(PAGE_KEYS.HOME, language)}
        component={Home}
      />
      <Route
        path={getInternalRoutePath(PAGE_KEYS.ABOUT, language)}
        component={ONas}
      />
      <Route
        path={getInternalRoutePath(PAGE_KEYS.OFFER, language)}
        component={Oferta}
      />
      <Route
        path={getInternalRoutePath(PAGE_KEYS.GALLERY, language)}
        component={Galeria}
      />
      <Route
        path={getInternalRoutePath(PAGE_KEYS.PRICING, language)}
        component={Cennik}
      />
      <Route
        path={getInternalRoutePath(PAGE_KEYS.CONTACT, language)}
        component={Kontakt}
      />
      <Route>
        <NotFound />
      </Route>
    </Switch>
  );
}

function AppMainContent() {
  return (
    <div className="min-h-screen flex flex-col">
      <Navbar />
      <main className="flex-1 pt-16">
        <ScrollRestoration />
        <AppRoutes />
      </main>
      <Footer />
      <ScrollToTopButton />
      <CookieConsent />
    </div>
  );
}

function LanguageAwareRouter() {
  const { language, isLanguageInitialized } = useLanguage();

  if (!isLanguageInitialized) {
    return null;
  }

  const base = language === defaultLang ? "/" : `/${language}`;
  const routerKey = `main-router-${base}`;

  return (
    <WouterRouter key={routerKey} base={base}>
      <LanguageRouterLogic />
      <AppMainContent />
    </WouterRouter>
  );
}

function App() {
  return (
    <HelmetProvider>
      <QueryClientProvider client={queryClient}>
        <LanguageProvider>
          <TooltipProvider>
            <Toaster />
            <LanguageAwareRouter />
          </TooltipProvider>
        </LanguageProvider>
      </QueryClientProvider>
    </HelmetProvider>
  );
}

export default App;

const LanguageContext = createContext<LanguageContextType | undefined>(
  undefined
);

export const LanguageProvider: React.FC<{ children: ReactNode }> = ({
  children,
}) => {
  const [language, setLanguageState] = useState<Language>(appDefaultLang);
  const [isLanguageInitialized, setIsLanguageInitialized] = useState(false);

  useEffect(() => {
    // Ustawiamy isLanguageInitialized na true zaraz po pierwszym renderowaniu providera.
    // To zapewnia, że komponenty zależne od tego stanu (jak LanguageAwareRouter)
    // będą mogły się poprawnie wyrenderować.
    setIsLanguageInitialized(true);
  }, []); // Pusta tablica zależności oznacza, że efekt uruchomi się tylko raz, po zamontowaniu.

  const setLanguage = (
    newLang: Language,
    options?: { preventNavigation?: boolean }
  ) => {
    setLanguageState(newLang);
    // Nie ma już potrzeby ustawiania isLanguageInitialized tutaj,
    // bo jest to robione w useEffect.
  };

  const t = (
    key: string,
    replacements?: Record<string, string | number>,
    fallback?: string
  ): string => {
    let translation = translations[language]?.[key];
    if (translation === undefined && fallback) {
      translation = fallback;
    } else if (translation === undefined) {
      translation = key;
    }

    if (replacements && typeof translation === "string") {
      Object.keys(replacements).forEach((placeholder) => {
        const regex = new RegExp(`{${placeholder}}`, "g");
        translation = translation.replace(
          regex,
          String(replacements[placeholder])
        );
      });
    }
    return translation || key;
  };

  return (
    <LanguageContext.Provider
      value={{ language, setLanguage, t, isLanguageInitialized }}
    >
      {children}
    </LanguageContext.Provider>
  );
};

export function useLanguage(): LanguageContextType {
  // Zmieniono na nazwaną funkcję
  const context = useContext(LanguageContext);
  if (!context) {
    throw new Error("useLanguage must be used within a LanguageProvider");
  }
  return context;
}

export const getLocalizedSlug = (pageKey: PageKey, lang: Language): string => {
  const slug = slugsConfig[lang]?.[pageKey];
  if (typeof slug === "string") return slug;
  const fallbackSlug = slugsConfig[defaultLang]?.[pageKey];
  if (typeof fallbackSlug === "string") return fallbackSlug;
  return pageKey.toLowerCase().replace(/_/g, "-");
};

export function getInternalRoutePath(pageKey: PageKey, lang: Language): string {
  if (pageKey === PAGE_KEYS.HOME) return "/";
  return `/${getLocalizedSlug(pageKey, lang)}`;
}

export function getLocalizedPath(
  pageKey: PageKey, // Typ jest już poprawny
  lang: Language,
  hashPageKey?: PageKey
): string {
  const langPrefix = lang === defaultLang ? "" : `/${lang}`;
  let basePathKey: PageKey = pageKey; // Inicjalizacja z poprawnym typem
  let actualHashKeyForSlug: PageKey | undefined = hashPageKey;

  // Sprawdzenia `startsWith` są na typie PageKey, który jest unią stringów literalnych, co jest OK
  if (pageKey.startsWith("OFFER_")) {
    basePathKey = PAGE_KEYS.OFFER;
    actualHashKeyForSlug = pageKey;
  } else if (pageKey.startsWith("GALLERY_PHOTO_")) {
    basePathKey = PAGE_KEYS.GALLERY;
    actualHashKeyForSlug = pageKey;
  }

  let baseSlugPart: string;
  if (basePathKey === PAGE_KEYS.HOME) {
    baseSlugPart = "";
  } else {
    baseSlugPart = `/${getLocalizedSlug(basePathKey, lang)}`;
  }

  let fullPath = `${langPrefix}${baseSlugPart}`;

  if (basePathKey === PAGE_KEYS.HOME && lang === defaultLang) {
    fullPath = "/";
  } else if (basePathKey === PAGE_KEYS.HOME && lang !== defaultLang) {
    // Dla strony głównej w innym języku, ścieżka to tylko prefiks językowy
    fullPath = `/${lang}`;
  } else if (
    basePathKey !== PAGE_KEYS.HOME &&
    fullPath === langPrefix &&
    slugPartIfHomePageWasEmpty(basePathKey, lang) === ""
  ) {
    // Obsługa przypadku, gdy slug dla strony innej niż HOME jest pusty (nie powinno się zdarzyć przy dobrej konfiguracji)
    // Jeśli slug jest pusty, a jest prefiks, to zostawiamy sam prefiks (np. /en)
    // Jeśli nie ma prefiksu (defaultLang), to zostawiamy "/"
    fullPath = langPrefix || "/";
  }

  if (actualHashKeyForSlug) {
    const hashSlugValue = getLocalizedSlug(actualHashKeyForSlug, lang);
    if (hashSlugValue) fullPath += `#${hashSlugValue}`;
  }
  return fullPath;
}

// Funkcja pomocnicza do getLocalizedPath
function slugPartIfHomePageWasEmpty(pageKey: PageKey, lang: Language): string {
  if (pageKey === PAGE_KEYS.HOME) return "";
  return `/${getLocalizedSlug(pageKey, lang)}`;
}

export function findPageKeyByLocalizedPath(currentFullPath: string): {
  pageKey: PageKey | null;
  lang: Language;
  hashKey: PageKey | null;
  hashSlug: string | null;
  basePathWithoutLang: string;
} {
  // console.log(`%c[FPLP v12.6] Input currentFullPath: "${currentFullPath}"`, "color: blue;");
  const [pathAndQuery, hashValueFromUrl] = currentFullPath.split("#");
  const pathOnly = pathAndQuery.split("?")[0];

  let detectedLang: Language = defaultLang; // Jawne typowanie
  let slugSegmentsForPathMatching: string[];

  const pathSegments = pathOnly.split("/").filter(Boolean);

  if (
    pathSegments.length > 0 &&
    supportedLngs.includes(pathSegments[0] as Language) // Type assertion jest OK tutaj
  ) {
    detectedLang = pathSegments[0] as Language; // Type assertion
    slugSegmentsForPathMatching = pathSegments.slice(1);
  } else {
    detectedLang = defaultLang;
    slugSegmentsForPathMatching = pathSegments;
  }

  const slugToAnalyze = slugSegmentsForPathMatching.join("/") || "";
  const basePathWithoutLang = `/${slugToAnalyze}`.replace(/\/+$/, "") || "/";

  let foundPageKey: PageKey | null = null;
  // Używamy `detectedLang` (które ma typ Language) do indeksowania slugsConfig
  const currentLangSlugs = slugsConfig[detectedLang];

  for (const keyCandidate in currentLangSlugs) {
    const key = keyCandidate as PageKey; // Type assertion
    if (
      Object.prototype.hasOwnProperty.call(currentLangSlugs, key) &&
      Object.values(PAGE_KEYS).includes(key) && // Sprawdź, czy klucz jest w PAGE_KEYS
      !key.startsWith("OFFER_") &&
      !key.startsWith("GALLERY_PHOTO_")
    ) {
      if (currentLangSlugs[key] === slugToAnalyze) {
        foundPageKey = key;
        break;
      }
    }
  }

  if (
    !foundPageKey &&
    slugToAnalyze === "" &&
    currentLangSlugs[PAGE_KEYS.HOME] === ""
  ) {
    foundPageKey = PAGE_KEYS.HOME;
  }

  if (!foundPageKey) {
    console.warn(
      `[FPLP v12.6] No pageKey found for slug "${slugToAnalyze}" (lang "${detectedLang}"). Path: "${currentFullPath}". Setting to NOT_FOUND.`
    );
    foundPageKey = PAGE_KEYS.NOT_FOUND;
  }

  let foundHashKey: PageKey | null = null;
  if (hashValueFromUrl) {
    for (const keyCandidate in currentLangSlugs) {
      const key = keyCandidate as PageKey; // Type assertion
      if (
        Object.prototype.hasOwnProperty.call(currentLangSlugs, key) &&
        Object.values(PAGE_KEYS).includes(key)
      ) {
        if (currentLangSlugs[key] === hashValueFromUrl) {
          foundHashKey = key;
          break;
        }
      }
    }
    if (!foundHashKey && foundPageKey !== PAGE_KEYS.NOT_FOUND) {
      console.warn(
        `[FPLP v12.6] No PageKey found for hashSlug "${hashValueFromUrl}" (lang "${detectedLang}"). Path: "${currentFullPath}".`
      );
    }
  }

  const langToReturn = detectedLang;

  // console.log(
  //   `%c  [FPLP v12.6] Output: pageKey="${foundPageKey}", lang="${langToReturn}", hashKey="${
  //     foundHashKey || "null"
  //   }", hashSlug="${
  //     hashValueFromUrl || null
  //   }", basePathWithoutLang="${basePathWithoutLang}"%c Path: "${currentFullPath}"`,
  //   "color: green;",
  //   "color: gray; font-style: italic;"
  // );
  return {
    pageKey: foundPageKey,
    lang: langToReturn,
    hashKey: foundHashKey,
    hashSlug: hashValueFromUrl || null,
    basePathWithoutLang: basePathWithoutLang,
  };
}

      "photo-tables-chairs-linens",

export const getLocalizedSlug = (pageKey: PageKey, lang: Language): string => {
  const slug = slugsConfig[lang]?.[pageKey];
  if (typeof slug === "string") return slug;
  const fallbackSlug = slugsConfig[defaultLang]?.[pageKey];
  if (typeof fallbackSlug === "string") {
    return fallbackSlug;
  }
  console.error(
    `[paths.ts] Critical: Slug for pageKey "${pageKey}" not found in lang "${lang}" NOR in default lang "${defaultLang}". Returning pageKey as slug.`
  );
  return pageKey.toLowerCase().replace(/_/g, "-");
};

export function getInternalRoutePath(pageKey: PageKey, lang: Language): string {
  if (pageKey === PAGE_KEYS.HOME) return "/";
  return `/${getLocalizedSlug(pageKey, lang)}`;
}

export function getLocalizedPath(
  pageKey: PageKey,
  lang: Language,
  hashPageKey?: PageKey
): string {
  const langPrefix = lang === defaultLang ? "" : `/${lang}`;
  let basePathKey: PageKey = pageKey;
  let actualHashKeyForSlug: PageKey | undefined = hashPageKey;

  if (pageKey.startsWith("OFFER_")) {
    basePathKey = PAGE_KEYS.OFFER;
    actualHashKeyForSlug = pageKey;
  } else if (pageKey.startsWith("GALLERY_PHOTO_")) {
    basePathKey = PAGE_KEYS.GALLERY;
    actualHashKeyForSlug = pageKey;
  }

  let baseSlugPart;
  if (basePathKey === PAGE_KEYS.HOME) {
    baseSlugPart = "";
  } else {
    baseSlugPart = `/${getLocalizedSlug(basePathKey, lang)}`;
  }

  let fullPath = `${langPrefix}${baseSlugPart}`;

  if (basePathKey === PAGE_KEYS.HOME && lang === defaultLang) {
    fullPath = "/";
  }
  if (
    basePathKey === PAGE_KEYS.HOME &&
    lang !== defaultLang &&
    fullPath === `/${lang}/`
  ) {
    fullPath = `/${lang}`;
  }
  if (basePathKey !== PAGE_KEYS.HOME && fullPath === "" && langPrefix) {
    fullPath = langPrefix;
  } else if (basePathKey !== PAGE_KEYS.HOME && fullPath === "") {
    fullPath = "/";
  }

  if (actualHashKeyForSlug) {
    const hashSlugValue = getLocalizedSlug(actualHashKeyForSlug, lang);
    if (hashSlugValue) fullPath += `#${hashSlugValue}`;
  }
  return fullPath;
}

export function findPageKeyByLocalizedPath(currentFullPath: string): {
  pageKey: PageKey | null;
  lang: Language;
  hashKey: PageKey | null;
  hashSlug: string | null;
  basePathWithoutLang: string;
} {
  // console.log(`%c[FPLP v12.5] Input currentFullPath: "${currentFullPath}"`, "color: blue;");
  const [pathAndQuery, hashValueFromUrl] = currentFullPath.split("#");
  const pathOnly = pathAndQuery.split("?")[0];

  let detectedLang: Language = defaultLang;
  let slugSegmentsForPathMatching: string[];

  const pathSegments = pathOnly.split("/").filter(Boolean);

  if (
    pathSegments.length > 0 &&
    supportedLngs.includes(pathSegments[0] as Language)
  ) {
    detectedLang = pathSegments[0] as Language;
    slugSegmentsForPathMatching = pathSegments.slice(1);
  } else {
    detectedLang = defaultLang;
    slugSegmentsForPathMatching = pathSegments;
  }

  const slugToAnalyze = slugSegmentsForPathMatching.join("/") || "";
  const basePathWithoutLang = `/${slugToAnalyze}`.replace(/\/+$/, "") || "/";

  let foundPageKey: PageKey | null = null;
  for (const keyCandidate in slugsConfig[detectedLang]) {
    const key = keyCandidate as PageKey;
    if (
      Object.prototype.hasOwnProperty.call(slugsConfig[detectedLang], key) &&
      Object.values(PAGE_KEYS).includes(key) &&
      !key.startsWith("OFFER_") &&
      !key.startsWith("GALLERY_PHOTO_")
    ) {
      if (slugsConfig[detectedLang][key] === slugToAnalyze) {
        foundPageKey = key;
        break;
      }
    }
  }
  if (
    !foundPageKey &&
    slugToAnalyze === "" &&
    slugsConfig[detectedLang][PAGE_KEYS.HOME] === ""
  ) {
    foundPageKey = PAGE_KEYS.HOME;
  }

  if (!foundPageKey) {
    console.warn(
      `[FPLP v12.5] No pageKey found for slug "${slugToAnalyze}" (lang "${detectedLang}"). Path: "${currentFullPath}". Setting to NOT_FOUND.`
    );
    foundPageKey = PAGE_KEYS.NOT_FOUND;
  }

  let foundHashKey: PageKey | null = null;
  if (hashValueFromUrl) {
    for (const keyCandidate in slugsConfig[detectedLang]) {
      const key = keyCandidate as PageKey;
      if (
        Object.prototype.hasOwnProperty.call(slugsConfig[detectedLang], key) &&
        Object.values(PAGE_KEYS).includes(key)
      ) {
        if (slugsConfig[detectedLang][key] === hashValueFromUrl) {
          foundHashKey = key;
          break;
        }
      }
    }
    if (!foundHashKey && foundPageKey !== PAGE_KEYS.NOT_FOUND) {
      console.warn(
        `[FPLP v12.5] No PageKey found for hashSlug "${hashValueFromUrl}" (lang "${detectedLang}"). Path: "${currentFullPath}".`
      );
    }
  }

  const langToReturn = detectedLang;

  // Zakomentowany domyślny log wyjściowy dla czystszej konsoli
  // console.log(
  //   `%c  [FPLP v12.5] Output: pageKey="${foundPageKey}", lang="${langToReturn}", hashKey="${
  //     foundHashKey || "null"
  //   }", hashSlug="${
  //     hashValueFromUrl || null
  //   }", basePathWithoutLang="${basePathWithoutLang}"%c Path: "${currentFullPath}"`,
  //   "color: green;",
  //   "color: gray; font-style: italic;"
  // );
  return {
    pageKey: foundPageKey,
    lang: langToReturn,
    hashKey: foundHashKey,
    hashSlug: hashValueFromUrl || null,
    basePathWithoutLang: basePathWithoutLang,
  };
}

function ScrollRestoration() {
  const [wouterLocation] = useLocation();

  useEffect(() => {
    const hashValue = window.location.hash.replace("#", "");
    if (!hashValue) {
      window.scrollTo(0, 0);
    } else {
      const scrollTimeoutId = setTimeout(() => {
        const element = document.getElementById(hashValue);
        if (element) {
          const navbarElement = document.querySelector("nav");
          const navbarHeight = navbarElement ? navbarElement.offsetHeight : 70;
          const yOffset = -(navbarHeight + 20);
          const y =
            element.getBoundingClientRect().top + window.pageYOffset + yOffset;
          window.scrollTo({ top: y, behavior: "smooth" });
        } else {
          if (LRL_DEBUG_MODE)
            console.warn(
              `[ScrollRestoration] Element with id "${hashValue}" not found.`
            );
        }
      }, 100);
      return () => clearTimeout(scrollTimeoutId);
    }
  }, [wouterLocation]);
  return null;
}

function LanguageRouterLogic() {
  const [wouterLocation, navigate] = useLocation();
  const {
    language: contextLanguage,
    setLanguage: setGlobalLanguage,
    isLanguageInitialized,
  } = useLanguage();

  const logicRunCounter = useRef(0);
  const userInitiatedLangChangeRef = useRef<Language | null>(null);

  const normalizePathForComparison = (p: string) => {
    const [pathPart, hashPart] = p.split("#");
    const normalizedPath = pathPart.replace(/\/+$/, "") || "/";
    return hashPart ? `${normalizedPath}#${hashPart}` : normalizedPath;
  };

  useEffect(() => {
    // USER_LANG_CHANGE_EFFECT
    if (!isLanguageInitialized) {
      // if (LRL_DEBUG_MODE) console.log(`%c[LRL_v13.5 USER_LANG_CHANGE_EFFECT] Waiting for init...`, "color: brown;");
      return;
    }

    if (userInitiatedLangChangeRef.current === contextLanguage) {
      return;
    }

    // if (LRL_DEBUG_MODE) console.log(`%c[LRL_v13.5 USER_LANG_CHANGE_EFFECT] Context language is now: "${contextLanguage}". Previous intent: "${userInitiatedLangChangeRef.current}"`, "color: orange; font-weight: bold;");
    userInitiatedLangChangeRef.current = contextLanguage;
    logicRunCounter.current = 0;

    const currentBrowserFullPath =
      window.location.pathname + window.location.search + window.location.hash;
    const {
      pageKey: currentPageKeyFromUrl,
      lang: langCurrentlyInUrl,
      hashKey: currentHashKeyFromUrl,
    } = findPageKeyByLocalizedPath(currentBrowserFullPath);

    // if (LRL_DEBUG_MODE) console.log(`  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] Current URL: "${currentBrowserFullPath}", langInUrl: "${langCurrentlyInUrl}", pageKeyFromUrl: "${currentPageKeyFromUrl}"`);

    if (langCurrentlyInUrl !== contextLanguage) {
      const targetPageKeyForNav =
        currentPageKeyFromUrl && currentPageKeyFromUrl !== PAGE_KEYS.NOT_FOUND
          ? currentPageKeyFromUrl
          : PAGE_KEYS.HOME;

      const newAbsolutePathTarget = getLocalizedPath(
        targetPageKeyForNav,
        contextLanguage,
        currentHashKeyFromUrl || undefined
      );

      let relativePathForNavigate = getInternalRoutePath(
        targetPageKeyForNav,
        contextLanguage
      );
      if (currentHashKeyFromUrl) {
        const hashSlug = getLocalizedSlug(
          currentHashKeyFromUrl,
          contextLanguage
        );
        if (hashSlug) relativePathForNavigate += `#${hashSlug}`;
      }

      const normalizedCurrent = normalizePathForComparison(
        currentBrowserFullPath
      );
      const normalizedNewTargetAbsolute = normalizePathForComparison(
        newAbsolutePathTarget
      );

      // if (LRL_DEBUG_MODE) console.log(`  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] New absolute target: "${newAbsolutePathTarget}", Relative for navigate: "${relativePathForNavigate}"`);

      if (normalizedCurrent !== normalizedNewTargetAbsolute) {
        // console.warn(
        //   `  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] ACTION: Navigating to reflect new context language. Target: "${newAbsolutePathTarget}" (using relative: "${relativePathForNavigate}")`
        // );
        navigate(relativePathForNavigate, { replace: true });
      } else {
        // if (LRL_DEBUG_MODE) console.log(`  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] URL already matches new context language. Clearing intent.`);
        userInitiatedLangChangeRef.current = null;
      }
    } else {
      // if (LRL_DEBUG_MODE) console.log(`  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] URL lang ("${langCurrentlyInUrl}") already matches new context language ("${contextLanguage}"). Clearing intent.`);
      userInitiatedLangChangeRef.current = null;
    }
  }, [contextLanguage, isLanguageInitialized, navigate]);

  useEffect(() => {
    // URL_SYNC_EFFECT
    if (!isLanguageInitialized) {
      return;
    }

    const currentBrowserFullPathForCheck =
      window.location.pathname + window.location.search + window.location.hash;
    const { lang: langInCurrentUrlCheck } = findPageKeyByLocalizedPath(
      currentBrowserFullPathForCheck
    );

    if (
      userInitiatedLangChangeRef.current === contextLanguage &&
      langInCurrentUrlCheck !== contextLanguage
    ) {
      // if (LRL_DEBUG_MODE) console.log(`%c[LRL_v13.5 URL_SYNC_EFFECT] User intent for lang "${contextLanguage}" is active, but URL lang is "${langInCurrentUrlCheck}". URL_SYNC waits.`, "color: gray;");
      return;
    }

    logicRunCounter.current += 1;
    const runId = logicRunCounter.current;
    const currentBrowserFullPath =
      window.location.pathname + window.location.search + window.location.hash;

    if (runId > 6) {
      console.error(
        `[LRL_v13.5 URL_SYNC_EFFECT #${runId}] FATAL LOOP. Path: "${currentBrowserFullPath}", ContextLang: "${contextLanguage}". Halting.`
      );
      logicRunCounter.current = 0;
      userInitiatedLangChangeRef.current = null;
      return;
    }

    const {
      pageKey: parsedPageKeyFromUrl,
      lang: langDetectedInUrl,
      hashKey: parsedHashKeyFromUrl,
    } = findPageKeyByLocalizedPath(currentBrowserFullPath);

    if (langDetectedInUrl !== contextLanguage) {
      console.warn(
        `  [LRL_v13.5 URL_SYNC_EFFECT #${runId}] ACTION (URL priority): Context lang ("${contextLanguage}") differs from URL lang ("${langDetectedInUrl}"). Updating context.`
      );
      setGlobalLanguage(langDetectedInUrl, { preventNavigation: true });
      return;
    }

    const pageKeyToUse =
      parsedPageKeyFromUrl && parsedPageKeyFromUrl !== PAGE_KEYS.NOT_FOUND
        ? parsedPageKeyFromUrl
        : PAGE_KEYS.HOME;
    const canonicalAbsolutePath = getLocalizedPath(
      pageKeyToUse,
      contextLanguage,
      parsedHashKeyFromUrl || undefined
    );
    const normalizedCurrent = normalizePathForComparison(
      currentBrowserFullPath
    );
    const normalizedCanonical = normalizePathForComparison(
      canonicalAbsolutePath
    );

    if (normalizedCurrent !== normalizedCanonical) {
      let reason = "Path MISMATCH";
      if (
        parsedPageKeyFromUrl === PAGE_KEYS.NOT_FOUND &&
        pageKeyToUse === PAGE_KEYS.HOME
      ) {
        reason = "Page NOT_FOUND, redirecting to HOME";
      }
      console.warn(
        `  [LRL_v13.5 URL_SYNC_EFFECT #${runId}] ACTION: ${reason}. Current: "${currentBrowserFullPath}", Canonical: "${canonicalAbsolutePath}". NAVIGATING.`
      );
      let relativePathForNavigate = getInternalRoutePath(
        pageKeyToUse,
        contextLanguage
      );
      if (parsedHashKeyFromUrl) {
        const hashSlug = getLocalizedSlug(
          parsedHashKeyFromUrl,
          contextLanguage
        );
        if (hashSlug) relativePathForNavigate += `#${hashSlug}`;
      }
      navigate(relativePathForNavigate, { replace: true });
      return;
    } else {
      logicRunCounter.current = 0;
      if (userInitiatedLangChangeRef.current === contextLanguage) {
        // if (LRL_DEBUG_MODE) console.log(`  [LRL_v13.5 URL_SYNC_EFFECT #${runId}] Path is canonical and lang intent fulfilled. Clearing userInitiatedLangChangeRef.`);
        userInitiatedLangChangeRef.current = null;
      }
    }
  }, [
    wouterLocation,
    contextLanguage,
    isLanguageInitialized,
    setGlobalLanguage,
    navigate,
  ]);

  return null;
}

function LanguageAwareRouter() {
  const { language, isLanguageInitialized } = useLanguage();

  const routerKey = `main-router-${base}`;

  return (
    <WouterRouter key={routerKey} base={base}>

      <AppMainContent />
    </WouterRouter>
  );
}

import { useState, useEffect } from "react";

} from "@/config/paths";

interface CategoryFilter {
  id: string;
  nameKey: string;
  pageKeyForHash: PageKey | null;
  count?: number;
  filterLogic: (attraction: Attraction) => boolean;
}

// Ta funkcja pomocnicza deriveInternalPathFromLocalized nie jest już potrzebna,
// jeśli WouterLink i navigate używają poprawnych ścieżek względem base

export default function Oferta() {
  const [activeCategory, setActiveCategory] = useState("all");
  const { t, language } = useLanguage();
  const [location, navigate] = useLocation(); // location to ścieżka względna do base

  const categories: CategoryFilter[] = [
    {
      id: "all",
      nameKey: "nav.allAttractions",
      pageKeyForHash: null,
      count: attractions.filter((attr) => attr.category !== "uslugi").length,

    },
    {
      id: "namioty",
      nameKey: "nav.tents",
      pageKeyForHash: PAGE_KEYS.OFFER_TENTS,

      id: "stoly-krzesla-obrusy",

      count: getAttractionsByCategory("wyposazenie").length,

      if (

        setActiveCategory(categoryFromHash.id);

    if (location !== pathForWouter) {

const LRL_DEBUG_MODE = false;

function ScrollRestoration() {
  const [wouterLocation] = useLocation();

  useEffect(() => {
    const hashValue = window.location.hash.replace("#", "");
    if (!hashValue) {
      window.scrollTo(0, 0);
    } else {
      const scrollTimeoutId = setTimeout(() => {
        const element = document.getElementById(hashValue);
        if (element) {
          const navbarElement = document.querySelector("nav");
          const navbarHeight = navbarElement ? navbarElement.offsetHeight : 70;
          const yOffset = -(navbarHeight + 20);
          const y =
            element.getBoundingClientRect().top + window.pageYOffset + yOffset;
          window.scrollTo({ top: y, behavior: "smooth" });
        } else {
          if (LRL_DEBUG_MODE)
            console.warn(
              `[ScrollRestoration] Element with id "${hashValue}" not found.`
            );
        }
      }, 100);
      return () => clearTimeout(scrollTimeoutId);
    }
  }, [wouterLocation]);
  return null;
}

function LanguageRouterLogic() {
  const [wouterLocation, navigate] = useLocation();
  const {
    language: contextLanguage,
    setLanguage: setGlobalLanguage,
    isLanguageInitialized,
  } = useLanguage();

  const logicRunCounter = useRef(0);
  const userInitiatedLangChangeRef = useRef<Language | null>(null);

  const normalizePathForComparison = (p: string) => {
    const [pathPart, hashPart] = p.split("#");
    const normalizedPath = pathPart.replace(/\/+$/, "") || "/";
    return hashPart ? `${normalizedPath}#${hashPart}` : normalizedPath;
  };

  useEffect(() => {
    if (!isLanguageInitialized) return;

    if (userInitiatedLangChangeRef.current === contextLanguage) {
      return;
    }

    userInitiatedLangChangeRef.current = contextLanguage;
    logicRunCounter.current = 0;

    const currentBrowserFullPath =
      window.location.pathname + window.location.search + window.location.hash;
    const {
      pageKey: currentPageKeyFromUrl,
      lang: langCurrentlyInUrl,
      hashKey: currentHashKeyFromUrl,
    } = findPageKeyByLocalizedPath(currentBrowserFullPath);

    if (langCurrentlyInUrl !== contextLanguage) {
      const targetPageKeyForNav =
        currentPageKeyFromUrl && currentPageKeyFromUrl !== PAGE_KEYS.NOT_FOUND
          ? currentPageKeyFromUrl
          : PAGE_KEYS.HOME;

      const newAbsolutePathTarget = getLocalizedPath(
        targetPageKeyForNav,
        contextLanguage,
        currentHashKeyFromUrl || undefined
      );

      let relativePathForNavigate = getInternalRoutePath(
        targetPageKeyForNav,
        contextLanguage
      );
      if (currentHashKeyFromUrl) {
        const hashSlug = getLocalizedSlug(
          currentHashKeyFromUrl,
          contextLanguage
        );
        if (hashSlug) relativePathForNavigate += `#${hashSlug}`;
      }

      const normalizedCurrent = normalizePathForComparison(
        currentBrowserFullPath
      );
      const normalizedNewTargetAbsolute = normalizePathForComparison(
        newAbsolutePathTarget
      );

      if (normalizedCurrent !== normalizedNewTargetAbsolute) {
        if (LRL_DEBUG_MODE)
          console.warn(
            `  [LRL_v13.5 USER_LANG_CHANGE_EFFECT] ACTION: Navigating to reflect new context language. Target: "${newAbsolutePathTarget}" (using relative: "${relativePathForNavigate}")`
          );
        navigate(relativePathForNavigate, { replace: true });
      } else {
        userInitiatedLangChangeRef.current = null;
      }
    } else {
      userInitiatedLangChangeRef.current = null;
    }
  }, [contextLanguage, isLanguageInitialized, navigate]);

  useEffect(() => {
    if (!isLanguageInitialized) return;

    const currentBrowserFullPathForCheck =
      window.location.pathname + window.location.search + window.location.hash;
    const { lang: langInCurrentUrlCheck } = findPageKeyByLocalizedPath(
      currentBrowserFullPathForCheck
    );

    if (
      userInitiatedLangChangeRef.current === contextLanguage &&
      langInCurrentUrlCheck !== contextLanguage
    ) {
      return;
    }

    logicRunCounter.current += 1;
    const runId = logicRunCounter.current;
    const currentBrowserFullPath =
      window.location.pathname + window.location.search + window.location.hash;

    if (runId > 6) {
      console.error(
        `[LRL_v13.5 URL_SYNC_EFFECT #${runId}] FATAL LOOP. Path: "${currentBrowserFullPath}", ContextLang: "${contextLanguage}". Halting.`
      );
      logicRunCounter.current = 0;
      userInitiatedLangChangeRef.current = null;
      return;
    }

    const {
      pageKey: parsedPageKeyFromUrl,
      lang: langDetectedInUrl,
      hashKey: parsedHashKeyFromUrl,
    } = findPageKeyByLocalizedPath(currentBrowserFullPath);

    if (langDetectedInUrl !== contextLanguage) {
      if (LRL_DEBUG_MODE)
        console.warn(
          `  [LRL_v13.5 URL_SYNC_EFFECT #${runId}] ACTION (URL has priority): Context lang ("${contextLanguage}") differs from URL lang ("${langDetectedInUrl}"). Updating context.`
        );
      setGlobalLanguage(langDetectedInUrl, { preventNavigation: true });
      return;
    }

    const pageKeyToUse =
      parsedPageKeyFromUrl && parsedPageKeyFromUrl !== PAGE_KEYS.NOT_FOUND
        ? parsedPageKeyFromUrl
        : PAGE_KEYS.HOME;
    const canonicalAbsolutePath = getLocalizedPath(
      pageKeyToUse,
      contextLanguage,
      parsedHashKeyFromUrl || undefined
    );
    const normalizedCurrent = normalizePathForComparison(
      currentBrowserFullPath
    );
    const normalizedCanonical = normalizePathForComparison(
      canonicalAbsolutePath
    );

    if (normalizedCurrent !== normalizedCanonical) {
      let reason = "Path MISMATCH";
      if (
        parsedPageKeyFromUrl === PAGE_KEYS.NOT_FOUND &&
        pageKeyToUse === PAGE_KEYS.HOME
      ) {
        reason = "Page NOT_FOUND, redirecting to HOME";
      }
      if (LRL_DEBUG_MODE)
        console.warn(
          `  [LRL_v13.5 URL_SYNC_EFFECT #${runId}] ACTION: ${reason}. Current: "${currentBrowserFullPath}", Canonical: "${canonicalAbsolutePath}". NAVIGATING.`
        );
      let relativePathForNavigate = getInternalRoutePath(
        pageKeyToUse,
        contextLanguage
      );
      if (parsedHashKeyFromUrl) {
        const hashSlug = getLocalizedSlug(
          parsedHashKeyFromUrl,
          contextLanguage
        );
        if (hashSlug) relativePathForNavigate += `#${hashSlug}`;
      }
      navigate(relativePathForNavigate, { replace: true });
      return;
    } else {
      logicRunCounter.current = 0;
      if (userInitiatedLangChangeRef.current === contextLanguage) {
        userInitiatedLangChangeRef.current = null;
      }
    }
  }, [
    wouterLocation,
    contextLanguage,
    isLanguageInitialized,
    setGlobalLanguage,
    navigate,
  ]);

  return null;
}

function LanguageAwareRouter() {
  const { language, isLanguageInitialized } = useLanguage();

  if (!isLanguageInitialized) {
    return null;
  }

  const base = language === defaultLang ? "/" : `/${language}`;
  const routerKey = `main-router-${base}`;

  return (
    <WouterRouter key={routerKey} base={base}>
      <LanguageRouterLogic />
      <AppMainContent />
    </WouterRouter>
  );
}
