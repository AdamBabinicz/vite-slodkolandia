import { Switch, Route } from "wouter";
import { QueryClientProvider } from "@tanstack/react-query";
import { HelmetProvider } from "react-helmet-async";
import { queryClient } from "./lib/queryClient";
import { Toaster } from "@/components/ui/toaster";
import { TooltipProvider } from "@/components/ui/tooltip";
import Navbar from "./components/Navbar";
import Footer from "./components/Footer";
import ScrollToTopButton from "./components/ScrollToTopButton";
import CookieConsent from "./components/CookieConsent";
import Home from "./pages/Home";
import Oferta from "./pages/Oferta";
import Galeria from "./pages/Galeria";
import Cennik from "./pages/Cennik";
import Kontakt from "./pages/Kontakt";
import ONas from "./pages/ONas";
import NotFound from "./pages/not-found";
import { LanguageProvider } from "@/hooks/useLanguage";

function Router() {
  return (
    <div className="min-h-screen flex flex-col">
      <Navbar />
      <main className="flex-1">
        <Switch>
          <Route path="/" component={Home} />
          <Route path="/o-nas" component={ONas} />
          <Route path="/oferta" component={Oferta} />
          <Route path="/galeria" component={Galeria} />
          <Route path="/cennik" component={Cennik} />
          <Route path="/kontakt" component={Kontakt} />
          <Route component={NotFound} />
        </Switch>
      </main>
      <Footer />
      <ScrollToTopButton />
      <CookieConsent />
    </div>
  );
}

function App() {
  return (
    <HelmetProvider>
      <QueryClientProvider client={queryClient}>
        <LanguageProvider>
        <TooltipProvider>
          <Toaster />
          <Router />
        </TooltipProvider>
        </LanguageProvider>
      </QueryClientProvider>
    </HelmetProvider>
  );
}

export default App;

const LanguageContext = createContext<LanguageContextType | undefined>(
  undefined
);

export const LanguageProvider: React.FC<{ children: ReactNode }> = ({
  children,
}) => {
  const [language, setLanguageState] = useState<Language>("pl");

  const setLanguage = (lang: Language) => {
    setLanguageState(lang);
  };

  const t = (
    key: string,
    replacements?: Record<string, string | number>
  ): string => {
    let translation = translations[language]?.[key] || key;
    if (replacements) {
      Object.keys(replacements).forEach((placeholder) => {
        const regex = new RegExp(`{${placeholder}}`, "g");
        translation = translation.replace(
          regex,
          String(replacements[placeholder])
        );
      });
    }
    return translation;
  };

  return (
    <LanguageContext.Provider value={{ language, setLanguage, t }}>
      {children}
    </LanguageContext.Provider>
  );
};

export const useLanguage = (): LanguageContextType => {
  const context = useContext(LanguageContext);
  if (!context) {
    throw new Error("useLanguage must be used within a LanguageProvider");
  }
  return context;
};

const LanguageContext = createContext<LanguageContextType | undefined>(
  undefined
);

export const LanguageProvider: React.FC<{ children: ReactNode }> = ({
  children,
}) => {
  const [language, setLanguageState] = useState<Language>("pl");

  const setLanguage = (lang: Language) => {
    setLanguageState(lang);
  };

  const t = (
    key: string,
    replacements?: Record<string, string | number>
  ): string => {
    let translation = translations[language]?.[key] || key;
    if (replacements) {
      Object.keys(replacements).forEach((placeholder) => {
        const regex = new RegExp(`{${placeholder}}`, "g");
        translation = translation.replace(
          regex,
          String(replacements[placeholder])
        );
      });
    }
    return translation;
  };

  return (
    <LanguageContext.Provider value={{ language, setLanguage, t }}>
      {children}
    </LanguageContext.Provider>
  );
};

export const useLanguage = (): LanguageContextType => {
  const context = useContext(LanguageContext);
  if (!context) {
    throw new Error("useLanguage must be used within a LanguageProvider");
  }
  return context;
};

      console.log("%cMutationObserver callback triggered", "color: orange;");
      let shouldAttemptOverride = false;
      let reasonForOverride = "";

      for (const mutation of mutationsList) {
        if (mutation.type === "attributes") {
          const lockAttributeAddedOrPresent =
            body.hasAttribute("data-scroll-locked") ||
            body.hasAttribute("data-radix-scroll-lock");

          if (
            mutation.attributeName === "data-scroll-locked" ||
            mutation.attributeName === "data-radix-scroll-lock"
          ) {
            if (lockAttributeAddedOrPresent) {
              console.log(
                `%cLock attribute (${mutation.attributeName}) detected or changed. Current body.style.overflow: "${body.style.overflow}"`,
                "color: blue;"
              );
              // Je≈õli overflow jest ju≈º hidden, chcemy nadpisaƒá. Je≈õli nie jest, poczekamy na zmianƒô stylu.
              if (body.style.overflow === "hidden") {
                shouldAttemptOverride = true;
                reasonForOverride = `Lock attribute (${mutation.attributeName}) present and overflow is hidden.`;
              } else {
                // Atrybut jest, ale overflow jeszcze nie 'hidden'.
                // Mo≈ºemy za≈Ço≈ºyƒá, ≈ºe zaraz bƒôdzie, albo czekaƒá na mutacjƒô 'style'.
                // Dla bezpiecze≈Ñstwa, poczekajmy na mutacjƒô 'style' lub sprawd≈∫my ponownie po kr√≥tkim timeout.
              }
            } else {
              console.log(
                `%cLock attribute (${mutation.attributeName}) removed. Radix should handle overflow restoration.`,
                "color: blue;"
              );
            }
          }

          if (mutation.attributeName === "style") {
            console.log(
              `%cStyle attribute changed. Current body.style.overflow: "${body.style.overflow}"`,
              "color: green;"
            );
            if (
              body.style.overflow === "hidden" &&
              lockAttributeAddedOrPresent
            ) {
              shouldAttemptOverride = true;
              reasonForOverride =
                "body.style.overflow became 'hidden' while a lock attribute was present.";
            } else if (
              body.style.overflow !== "hidden" &&
              body.style.overflow !== "visible" &&
              lockAttributeAddedOrPresent
            ) {
              // Je≈õli Radix zmieni≈Ç na co≈õ innego ni≈º visible, a powinien byƒá visible
              // console.log(`%cOverflow is "${body.style.overflow}" but should be visible due to lock.`, 'color: red');
            }
          }
        }
      }

      if (shouldAttemptOverride) {
        if (body.style.overflow !== "visible") {
          // Nadpisuj tylko je≈õli faktycznie jest 'hidden' lub co≈õ innego
          console.log(
            `%c!!! ATTEMPTING OVERRIDE. Reason: ${reasonForOverride} Setting overflow: visible !important.`,
            "color: red; font-weight: bold;"
          );
          body.style.setProperty("overflow", "visible", "important");
          setTimeout(
            () =>
              console.log(
                "%cAfter override attempt, body.style.overflow:",
                "color: purple;",
                body.style.overflow
              ),
            0
          );
        } else {
          // console.log('%cOverride condition met, but overflow is already visible. No action.', 'color: grey;');
        }
      }
    });

    observer.observe(body, {
      attributes: true,
      attributeFilter: [
        "style",
        "data-scroll-locked",
        "data-radix-scroll-lock",
      ],
    });

    const initialLockAttributePresent =
      body.hasAttribute("data-scroll-locked") ||
      body.hasAttribute("data-radix-scroll-lock");
    if (initialLockAttributePresent && body.style.overflow === "hidden") {
      console.log(
        "%cInitial state: body is locked and overflow is hidden. Overriding on mount.",
        "color: red; font-weight: bold;"
      );
      body.style.setProperty("overflow", "visible", "important");
    }

    return () => {
      console.log("%cMutationObserver disconnected", "color: orange;");
      observer.disconnect();
    };
  }, []);

  return (
    <HelmetProvider>
      <QueryClientProvider client={queryClient}>
        <LanguageProvider>
          <TooltipProvider>
            <Toaster />
            <Router />
          </TooltipProvider>
        </LanguageProvider>
      </QueryClientProvider>
    </HelmetProvider>
  );
}

export default App;

export type Language = "pl" | "en";
export type PageKey =
  | "HOME"
  | "ABOUT"
  | "OFFER"
  | "GALLERY"
  | "PRICING"
  | "CONTACT"
  | "PRIVACY_POLICY"
  | "TERMS";

export type OfferHashKey =
  | "TENTS"
  | "INFLATABLES"
  | "POPCORN"
  | "COTTON_CANDY"
  | "CHOCOLATE_FOUNTAIN";

export type GalleryHashKey =
  | "PHOTO_TENTS"
  | "PHOTO_INFLATABLES"
  | "PHOTO_COTTON_CANDY"
  | "PHOTO_POPCORN"
  | "PHOTO_FOUNTAINS";

export type HashKey = OfferHashKey | GalleryHashKey;

const basePaths: Record<PageKey, Record<Language, string>> = {
  HOME: { pl: "/", en: "/" },
  ABOUT: { pl: "/o-nas", en: "/about-us" },
  OFFER: { pl: "/oferta", en: "/offer" },
  GALLERY: { pl: "/galeria", en: "/gallery" },
  PRICING: { pl: "/cennik", en: "/pricing" },
  CONTACT: { pl: "/kontakt", en: "/contact" },
  PRIVACY_POLICY: { pl: "/polityka-prywatnosci", en: "/privacy-policy" },
  TERMS: { pl: "/regulamin", en: "/terms-of-service" },
};

const hashSlugs: Record<HashKey, Record<Language, string>> = {
  // Oferta
  TENTS: { pl: "namioty", en: "tents" },
  INFLATABLES: { pl: "dmuchance", en: "inflatables" },
  POPCORN: { pl: "popcorn", en: "popcorn" },
  COTTON_CANDY: { pl: "wata-cukrowa", en: "cotton-candy" },
  CHOCOLATE_FOUNTAIN: { pl: "fontanna-czekoladowa", en: "chocolate-fountain" },
  // Galeria
  PHOTO_TENTS: { pl: "foto-namioty", en: "photo-tents" },
  PHOTO_INFLATABLES: { pl: "foto-dmuchance", en: "photo-inflatables" },
  PHOTO_COTTON_CANDY: { pl: "foto-wata", en: "photo-cotton-candy" }, // Upewnij siƒô, ≈ºe nazwa klucza pasuje do Navbar
  PHOTO_POPCORN: { pl: "foto-popcorn", en: "photo-popcorn" },
  PHOTO_FOUNTAINS: { pl: "foto-fontanny", en: "photo-fountains" }, // Upewnij siƒô, ≈ºe nazwa klucza pasuje do Navbar
};

export function getBasePath(pageKey: PageKey, lang: Language): string {
  return (
    basePaths[pageKey]?.[lang] || basePaths[pageKey]?.["pl"] || "/not-found"
  );
}

export function getHashSlug(hashKey: HashKey, lang: Language): string {
  return hashSlugs[hashKey]?.[lang] || hashSlugs[hashKey]?.["pl"] || "";
}

export function getLocalizedPath(
  pageKey: PageKey,
  lang: Language,
  hashKey?: HashKey
): string {
  let path = getBasePath(pageKey, lang);
  if (hashKey) {
    const slug = getHashSlug(hashKey, lang);
    if (slug) {
      path += `#${slug}`;
    }
  }
  return path;
}

export function findPageKeyByPath(currentPath: string): {
  pageKey: PageKey | null;
  lang: Language | null;
  hash: string | null;
} {
  const [pathOnly, hashValue] = currentPath.split("#");

  for (const lang of ["pl", "en"] as Language[]) {
    for (const key in basePaths) {
      const pageKey = key as PageKey;
      if (basePaths[pageKey][lang] === pathOnly) {
        return { pageKey, lang, hash: hashValue || null };
      }
    }
  }
  return { pageKey: null, lang: null, hash: null };
}

function ScrollRestoration() {
  const [location] = useLocation();

  useEffect(() => {
    if (!location.includes("#")) {
      window.scrollTo(0, 0);
    } else {
      const hash = location.split("#")[1];
      if (hash) {
        setTimeout(() => {
          // Daj czas na renderowanie DOM
          const element = document.getElementById(hash);
          if (element) {
            const navbarHeight =
              document.querySelector("nav")?.offsetHeight || 70; // Domy≈õlna wysoko≈õƒá
            const yOffset = -navbarHeight - 20;
            const y =
              element.getBoundingClientRect().top +
              window.pageYOffset +
              yOffset;
            window.scrollTo({ top: y, behavior: "smooth" });
          }
        }, 100); // Kr√≥tkie op√≥≈∫nienie
      }
    }
  }, [location]);

  return null;
}

function AppRoutes() {
  const [wouterLocation, setWouterLocation] = useLocation();
  const { language, setLanguage: setGlobalLanguage } = useLanguage();

  useEffect(() => {
    const currentFullUrl = window.location.pathname + window.location.hash;
    const {
      pageKey: detectedPageKey,
      lang: detectedLang,
      hash: detectedHashValue,
    } = findPageKeyByPath(currentFullUrl);

    if (detectedPageKey) {
      if (detectedLang && detectedLang !== language) {
        setGlobalLanguage(detectedLang);
      }

      const canonicalPathForWouter = basePaths[detectedPageKey]["pl"]; // Kanoniczna ≈õcie≈ºka (np. polska) dla komponent√≥w Route
      let wouterShouldBePath = canonicalPathForWouter;

      if (detectedHashValue) {
        // Tutaj wa≈ºne jest, aby hash dla woutera by≈Ç te≈º w kanonicznej (polskiej) formie, je≈õli hashe sƒÖ lokalizowane
        // Je≈õli hashe nie sƒÖ lokalizowane lub zawsze u≈ºywasz polskiego hasha dla woutera:
        wouterShouldBePath += `#${detectedHashValue}`;
      }

      // Je≈õli aktualna ≈õcie≈ºka woutera nie jest kanonicznƒÖ, a URL w przeglƒÖdarce jest poprawny (zlokalizowany)
      // to musimy zaktualizowaƒá wewnƒôtrznƒÖ ≈õcie≈ºkƒô woutera, aby dopasowaƒá komponent
      if (wouterLocation.split("#")[0] !== canonicalPathForWouter) {
        // Nie zmieniamy URL w przeglƒÖdarce, bo jest ju≈º zlokalizowany (lub w≈Ça≈õnie zosta≈Ç ustawiony przez setLanguage)
        // Zmieniamy tylko to, co "widzi" wouter
        setWouterLocation(wouterShouldBePath, { replace: true });
      }
    } else if (
      wouterLocation !== "/" &&
      !Object.values(basePaths).some((paths) =>
        Object.values(paths).includes(wouterLocation.split("#")[0])
      )
    ) {
      // Je≈õli ≈õcie≈ºka woutera nie pasuje do ≈ºadnej znanej bazowej ≈õcie≈ºki (ani polskiej ani angielskiej)
      // i nie jest to "/", to prawdopodobnie jest to 404 z perspektywy definicji ≈õcie≈ºek
      // Mo≈ºna by tu przekierowaƒá na zlokalizowany 404 lub g≈Ç√≥wnƒÖ stronƒô
    }
  }, [wouterLocation, language, setGlobalLanguage, setWouterLocation]);

  // Definicje ≈õcie≈ºek kanonicznych (np. polskich)
  // Komponenty stron (Home, ONas, etc.) same w sobie powinny u≈ºywaƒá hooka useLanguage()
  // do pobierania tre≈õci w odpowiednim jƒôzyku.
  return (
    <Switch>
      <Route path={basePaths["HOME"]["pl"]} component={Home} />
      <Route path={basePaths["ABOUT"]["pl"]} component={ONas} />
      <Route path={basePaths["OFFER"]["pl"]} component={Oferta} />
      <Route path={basePaths["GALLERY"]["pl"]} component={Galeria} />
      <Route path={basePaths["PRICING"]["pl"]} component={Cennik} />
      <Route path={basePaths["CONTACT"]["pl"]} component={Kontakt} />
      {/*
      Je≈õli masz strony dla polityki prywatno≈õci i regulaminu:
      <Route path={basePaths["PRIVACY_POLICY"]["pl"]} component={PrivacyPolicyPage} />
      <Route path={basePaths["TERMS"]["pl"]} component={TermsOfServicePage} />
      Pamiƒôtaj o dodaniu odpowiednich PageKey i ≈õcie≈ºek w paths.ts
      */}
      <Route>
        <NotFound />
      </Route>
    </Switch>
  );
}
